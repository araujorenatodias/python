import matplotlib.pyplot as plt #importa a biblioteca para a plotagem a = ["19/02","19/03","19/04","19/05","19/06"] #cria array com os valores b = [5.2,3.9,3,2.5,3] b2 = [3,3,2,1,1] plt.bar(a,b, color = 'blue') #faz a plotagem da barra bar(x,y) plt.bar(a,b2, color = 'yellow') #faz a plotagem da barra bar(x,y) plt.ylabel('Reais (R$)') # Label do eixo Y plt.xlabel('Data') # Label do eixo X plt.show() # exibe a plotagem na tela
